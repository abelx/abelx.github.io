### 快排
复杂度

- 时间：nlogn
- 空间：1

不稳定

```
class QuickSorter: public Sorter {
public:
    void sort(vector<int> &v) {
        if(v.size() <= 0) return;
        part(v, 0, v.size()-1);
    }
private:
    void part(vector<int> &v, int left, int right) {
        if(right-left<1) return;
        int tmp = v[left], i=left, j=right;
        while(i<j) {
            while(j>i && v[j] >= tmp) j--;
            v[i] = v[j];
            while(j>i && v[i] <= tmp) i++;
            v[j] = v[i]; 
        }
        v[i] = tmp;

        part(v, left, i-1);
        part(v, i+1, right);
    }
};
```
### 堆排 & 堆

/*基于vector实现的大顶堆*/
class MaxHeap {
public:
    MaxHeap(vector<int> &v) {
        this->v = v;
        for(int i=v.size()/2 - 1; i>=0; i--) {
            tuneDown(i, v.size()-1);
        }
    }
    void push(int a) {
        v.push_back(a);
        tuneUp(v.size()-1);
    }
    void pop() {
        if(empty()) return;
        swap(v.front(), v.back());
        v.pop_back();
        tuneDown(0, v.size()-1);
    }
    int top() {
        return v[0];
    }
    bool empty() {
        return v.size() < 1;
    }
    
private:
    void tuneDown(int sIdx, int eIdx) {
        int root = sIdx, maxIdx;
        while(root < eIdx) {
            int left = 2*root+1;
            if(left > eIdx) return;
            int right = 2*root+2;
            if(right <= eIdx &&  v[right] > v[left]) maxIdx = right;
            else maxIdx =  left;
            if(v[root] >= v[maxIdx]) return;
            swap(v[root], v[maxIdx]);
            root = maxIdx;
        }
    }
    void tuneUp(int idx) {
        while(idx > 0) {
            int p = (idx-1)/2;
            if(v[p] > v[idx]) return;
            swap(v[p], v[idx]);
            idx = p;
        }
    }
    
    vector<int> v;
};

/*空间复杂度为o(1)的堆排序*/
class HeapSorter: public Sorter {
public:
    void sort(vector<int> &v) {
        constructMaxHeap(v);
        
        for(int i=v.size()-1; i>0; i--) {
            swap(v[0], v[i]);
            tuneMaxHeap(v, 0, i-1);
        }
    }
private:
    void constructMaxHeap(vector<int> &v) {
        int s = v.size();
        
        for(int i=s/2 - 1; i>=0; i--) {
            tuneMaxHeap(v, i, s-1);
        }
    }
    
    /*自上而下调整堆*/
    void tuneMaxHeap(vector<int> &v, int start, int end) {
        int i = start;
        while(i < end) {
            int left = 2*i+1, right = 2*i+2, max_idx;
            if(left <= end) max_idx = left;
            else break;
            if(right <= end && v[right] > v[left]) max_idx = right;
            if(v[i] < v[max_idx]) {
                swap(v[i], v[max_idx]);
                i = max_idx;
            }
            else break;
        }
    }
};
```

### 归并
复杂度

- 时间：nlogn
- 空间：n

稳定

```
class MergeSorter: public Sorter {
public:
    void sort(vector<int> &v) {
        vTmp = v;
        _partSort(v, 0, v.size()-1);
    }

private:
    void _merge(vector<int> &v, int sIdx, int eIdx) {
        int mid = (sIdx+eIdx) / 2;
        int i=sIdx, j=sIdx, k=mid+1;
        while(i<=eIdx) {
            if(k>eIdx) vTmp[i++] = v[j++];
            else if(j>mid) vTmp[i++] = v[k++]; 
            else if(v[j] < v[k]) {
                vTmp[i++] = v[j++];
            }
            else {
                vTmp[i++] = v[k++];
            }
        }
        for(int i=sIdx; i<=eIdx; i++) {
            v[i] = vTmp[i];
        }      
    }
    void _partSort(vector<int> &v, int sIdx, int eIdx) {
        if(sIdx >= eIdx) return;
        int mid = (sIdx+eIdx) / 2;
        _partSort(v, sIdx, mid);
        _partSort(v, mid+1, eIdx);
        _merge(v, sIdx, eIdx);
    }
    
    vector<int> vTmp;
};
```
### 选择
复杂度

- 时间：n^2
- 空间：1

不稳定：因为在选择的时候要进行交换

```
class SelectSorter: public Sorter {
public:
    void sort(vector<int> &v) {
        for(int i=v.size()-1; i>0; i--) {
            int maxIdx = 0;
            for(int j=1; j<i; j++) {
                if(v[j] > v[maxIdx]) maxIdx = j;
            }
            if(v[i] < v[maxIdx]) swap(v[i], v[maxIdx]);
        }
    }
};
```
### 冒泡
复杂度

- 时间：n^2
- 空间：1

稳定

```
class BubSorter: public Sorter {
public:
    void sort(vector<int> &v) {
        for(int i=0; i<v.size()-1; i++) {
            for(int j=v.size()-1; j > i; j--) {
                if(v[j-1] > v[j]) swap(v[j-1], v[j]);
            }
        }
    }
};
